135 inline void TAPPrintListener::OnTestProgramEnd(const UnitTest& test)
136 {
137  detail::LogStream stream;
138 
139  int number = 1;
140  for( int k=0, count=test.total_test_case_count(); k < count; ++k )
141  {
142  const TestCase& test_case = *test.GetTestCase(k);
143 
144  OnReportTestCase(&stream, test_case, number);
145 
146  number += test_case.total_test_count();
147  }
148  detail::iuConsole::output("1..%d\n", number-1);
149 }
150 inline void TAPPrintListener::OnReportTestCase(detail::IOutStream* const stream, const TestCase& test_case, int top)
151 {
152  stream->Printf("# %s started.\n", test_case.name());
153 
154  for( int i=0, count=test_case.total_test_count(); i < count; ++i )
155  {
156  const TestInfo* test_info = test_case.GetTestInfo(i);
157  if( !test_info->should_run() )
158  {
159  // スキップ
160  stream->Printf("ok %d # SKIP - %s\n", top+i, test_info->name());
161  }
162  else
163  {
164  if( test_info->HasFailure() )
165  {
166  stream->Printf("not ok %d - %s ", top+i, test_info->name() );
167  for( int j=0, count=test_info->result()->total_part_count(); j < count; ++j )
168  {
169  const TestPartResult& part = test_info->result()->GetTestPartResult(j);
170  ::std::string summary = part.make_message();
171  // 改行を置き換え
172  detail::StringReplace(summary, '\n', ", ");
173  stream->Printf("%s. ", summary.c_str() );
174  }
175  stream->Printf("\n");
176  }
177  else
178  {
179  stream->Printf("ok %d - %s\n", top+i, test_info->name() );
180  }
181  }
182  }
183 
184  stream->Printf("# %s ended.\n", test_case.name());
185 }

187 inline void TAPFileGeneratorListener::OnTestProgramEnd(const UnitTest& test)
188 {
189  for( int k=0, count=test.total_test_case_count(); k < count; ++k )
190  {
191  const TestCase& test_case = *test.GetTestCase(k);
192 
193  IUTEST_UNUSED_VAR(test);
194  IFile* fp = detail::IFileSystem::New();
195  if( fp == NULL )
196  {
197  return;
198  }
199 
200  ::std::string filepath = m_output_path;
201  ::std::string name = test_case.name();
202  detail::StringReplace(name, '/', "_");
203  filepath += detail::GetPathSeparator();
204  filepath += name;
205  filepath += ".tap";
206 
207  if( !fp->Open(filepath.c_str(), IFile::OpenWrite) )
208  {
209  fprintf(stderr, "Unable to open file \"%s\".\n", filepath.c_str());
210  fflush(stderr);
211  }
212 
213  OnReportTestCase(fp, test_case);
214 
215  fp->Printf("1..%d\n", test_case.total_test_count());
216 
217  fp->Close();
218  detail::IFileSystem::Free(fp);
219  }
220 }
