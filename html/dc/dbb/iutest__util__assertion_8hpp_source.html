483 template<typename T1, typename T2>
484 ::iutest::AssertionResult IUTEST_ATTRIBUTE_UNUSED_ CmpHelperEqIterator(T1 b1, T1 e1, T2 b2, T2 e2)
485 {
486  int elem=0;
487  bool result = true;
488  ::iutest::Message ar;
489  for(elem=0; b1 != e1 && b2 != e2; ++b1, ++b2, ++elem)
490  {
491  if(!::iutest::internal::backward::EqHelper<false>::Compare("", "", *b1, *b2))
492  {
493  result = false;
494  ar << "\nMismatch in a position " << elem << ": "
495  << ::iutest::internal::FormatForComparisonFailureMessage(*b1, *b2)
496  << " vs " << ::iutest::internal::FormatForComparisonFailureMessage(*b2, *b1);
497  }
498  }
499  if(b1 != e1)
500  {
501  int elem1 = elem;
502  for(; b1 != e1; ++b1, ++elem1) {}
503  result = false;
504  ar << "\nMismatch element : " << elem1 << " vs " << elem;
505  }
506  if(b2 != e2)
507  {
508  int elem2 = elem;
509  for(; b2 != e2; ++b2, ++elem2) {}
510  result = false;
511  ar << "\nMismatch element : " << elem << " vs " << elem2;
512  }
513  if( !result )
514  {
515  return ::iutest::AssertionFailure() << ar;
516  }
517  return ::iutest::AssertionSuccess();
518 }
